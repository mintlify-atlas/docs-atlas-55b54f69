## Stock Visualization

When analyzing holdings, users see detailed charts:

```jsx
// From StockHistoryGraph.jsx:5-53
const StockHistoryGraph = ({ dataframe }) => {
  const [hasData, setHasData] = useState(false);
  
  useEffect(() => {
    if (dataframe !== null && dataframe.data) {
      setHasData(true);
      
      // Extract dates and prices from Prophet forecast
      const dx = dataframe.data.map((item) => item[0]);  // Dates
      const dy = dataframe.data.map((item) => item[18]); // Predictions (yhat)
      
      const plotData = [{
        x: dx,
        y: dy,
        type: "scatter",
        mode: "lines",
        marker: { color: "white" }
      }];
      
      const layout = {
        title: "Stock History",
        xaxis: { title: "Date" },
        yaxis: { title: "Price" },
        plot_bgcolor: "black",
        paper_bgcolor: "black",
        font: { color: "white" }
      };
      
      Plotly.newPlot("stock-graph", plotData, layout);
    }
  }, [dataframe]);
  
  return (
    <>
      {!hasData ? (
        <div className="flex text-white p-4 m-4 rounded-lg relative bottom-24 bg-black opacity-50">
          <p>Ask a query to generate results</p>
        </div>
      ) : (
        <div id="stock-graph" className="flex text-white p-4 m-4 rounded-lg relative bottom-24 bg-black opacity-50"></div>
      )}
    </>
  );
};
```

# Conceptual schema for portfolio management
class Portfolio(Base):
    __tablename__ = "portfolios"
    
    id = Column(Integer, primary_key=True)
    user_id = Column(String, ForeignKey("users.id"))
    ticker = Column(String)  # e.g., "AAPL"
    quantity = Column(Integer)
    purchase_price = Column(Float)
    purchase_date = Column(DateTime)
    
class Transaction(Base):
    __tablename__ = "transactions"
    
    id = Column(Integer, primary_key=True)
    user_id = Column(String, ForeignKey("users.id"))
    ticker = Column(String)
    action = Column(Enum("BUY", "SELL"))
    quantity = Column(Integer)
    price = Column(Float)
    timestamp = Column(DateTime, default=datetime.now)

## Performance Metrics (Planned)

Future versions could calculate:

```python
# Conceptual performance calculations
def calculate_portfolio_metrics(holdings):
    total_value = sum(h.quantity * get_current_price(h.ticker) for h in holdings)
    total_cost = sum(h.quantity * h.purchase_price for h in holdings)
    
    absolute_return = total_value - total_cost
    percentage_return = (absolute_return / total_cost) * 100
    
    return {
        "total_value": total_value,
        "total_cost": total_cost,
        "absolute_return": absolute_return,
        "percentage_return": percentage_return,
        "top_performer": max(holdings, key=lambda h: h.current_return),
        "worst_performer": min(holdings, key=lambda h: h.current_return)
    }
```
